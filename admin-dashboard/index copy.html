/* --- NEWSLETTER SUBSCRIPTION LOGIC --- */ async function fetchSubscribers(silent = false) { try { const { data, error } = await supabaseInstance.from('subscribers').select('*').order('subscribed_at', { ascending: false }); if (error) throw error; allSubscribers = data || []; renderSubscribersTable(); } catch (err) { console.error("Failed to load subscribers:", err); if (!silent) showErrorToast("Failed to load subscribers."); document.getElementById('subscribersTableBody').innerHTML = `Failed to load subscribers.`; } } function renderSubscribersTable() { const tbody = document.getElementById('subscribersTableBody'); if (allSubscribers.length === 0) { tbody.innerHTML = `No active subscribers found.`; return; } tbody.innerHTML = allSubscribers.map(sub => { const statusHtml = sub.confirmed ? 'Active' : 'Pending'; return `
${sub.email}
${new Date(sub.subscribed_at).toLocaleDateString()} ${statusHtml}
`; }).join(''); lucide.createIcons(); } function confirmDeleteSubscriber(id) { openConfirmModal("Delete Subscriber", "Are you sure you want to permanently remove this email from your mailing list?", true, async () => { showLoader('loadingData'); try { const { error } = await supabaseInstance.from('subscribers').delete().eq('id', id); if (error) throw error; showToast("Subscriber removed successfully."); fetchSubscribers(true); } catch (err) { showErrorToast(err.message); } finally { hideLoader(); } }); } function openNewsletterModal() { document.getElementById('newsletterComposeForm').reset(); document.getElementById('newsletterModal').classList.remove('modal-hidden'); } function closeNewsletterModal() { document.getElementById('newsletterModal').classList.add('modal-hidden'); } async function sendNewsletter() { const subject = document.getElementById('newsSubject').value.trim(); const htmlContent = document.getElementById('newsHtml').value.trim(); if (!subject || !htmlContent) { showErrorToast("Please fill out both subject and content."); return; } const btn = document.getElementById('btnSendNewsletter'); btn.disabled = true; const originalBtnText = btn.innerHTML; btn.innerHTML = ' Sending...'; lucide.createIcons(); showLoader('loadingData'); try { const { error } = await supabaseInstance.rpc('send_newsletter', { email_subject: subject, html_content: htmlContent }); if (error) throw error; showToast("Newsletter dispatched successfully to all subscribers!"); closeNewsletterModal(); } catch (err) { console.error("Newsletter error:", err); showErrorToast(err.message || "Failed to send newsletter."); } finally { btn.disabled = false; btn.innerHTML = originalBtnText; hideLoader(); } }
